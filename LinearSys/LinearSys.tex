   连续周期信号：
   $$	f(t)=f(t+mT),m=0,\pm 1,\pm 2,\dots$$ 
   离散周期信号：
   $$	f(k)=f(k+mN),m=0,\pm 1,\pm 2,\dots$$ 
\end{definition}

\end{lemma}
